-- CRUD

select * from user_role;

insert into user_role(role_name)
	values('test_role');
	
update user_role
	set role_name = 'test_user_role'
	where role_name = 'test_role';
	
delete from user_role
	where role_name = 'test_user_role';


-- ORDER BY

select product_name, product_price 
	from product
	order by product_price desc;


-- MIN/MAX

select max(product_price)
	from product;
	
select min(product_price)
	from product;
	

-- COUNT/AVG/SUM

select count(product_name)
	from product
	where product_name = 'TBM 960';
	
select avg(product_price)
	from product;

select sum(product_price)
	from product;
	

-- LIKE

select count(product_name)
	from product
	where product_name like 'CESSNA%';
	

select product_name,product_price
	from product
	where product_name like 'CESSNA%';
